[docs]    def set_voltage_unit(self, unit: str):
        """
        Select the units for output amplitude (does not affect offset voltage or
        high/low levels). The default is VPP.

        .. note::
            The output units for amplitude cannot be set to dBm if the output
            termination is currently set to “high impedance”.

        :param unit: Output type of VPP or VRMS or DBM.
        :type unit: str

        """
        unit = unit.upper()
        if unit != "VPP" or unit != "VRMS" or unit != "DBM":
            raise exceptions.MalformedParamterException(unit,
                                                        ["VPP", "VRMS", "DBM"])
        self.instrument.write(f"VOLTage:UNIT {unit}")

[docs]    def set_output_load(self, load: float):
        """
        Select the desired output termination (i.e., the impedance of the load
        attached to the output of the Agilent 33210A). The specified value is used
        for amplitude, offset, and high/low level settings. You can set the load to
        any value from 1Ω to 10 kΩ.

        :rtype: float
        :raises: MalformedParameterException
        """
        min_output_load = float(self.instrument.query("OUTPut? MINimum"))
        max_output_load = float(self.instrument.query("OUTPut? MAXimum"))
        if (load < min_output_load) or (load > max_output_load):
            raise exceptions.MalformedParamterException(load,
                                                        f"{min_output_load} - {max_output_load}")
        self.instrument.write(f"OUTPut:LOAD {load}")

[docs]    def set_output_polarity(self, polarity: str):
        """
        In the normal mode (default), the waveform goes positive during the first part of the cycle.
        In the inverted mode, the waveform goes negative during the first part of
        the cycle.

        :param polarity: Polarity of signal of NORM or INV
        :type polarity: str
        :raises: MalformedParameterException
        """
        polarity = polarity.upper()
        if polarity == "NORM":
            polarity = "NORMal"
        elif polarity == "INV":
            polarity = "INVerted"
        else:
            raise exceptions.MalformedParamterException(polarity, ["NORM", "INV"])
        self.instrument.write(f"OUTPut:POLarity {polarity}")

[docs]    def set_function(self, function: str):
        """
        Select the output function. The selected waveform is output using the
        previously selected frequency, amplitude, and offset voltage settings.

        :param function: Type of function with options of SIN, SQU, RAMP, PULS
            NOIS, DC
        :type function: str
        :raises: MalformedParameterException
        """
        function = function.upper()
        functions = {"SIN": "SINe",
                     "SQU": "SQUare",
                     "RAMP": "RAMP",
                     "PULS": "PULSe",
                     "NOIS": "NPISe",
                     "DC": "DC"}
        try:
            function = functions[function]
            self.instrument.write(f"FUNCtion {function}")
        except KeyError:
            raise exceptions.MalformedParamterException(function, list(functions.keys()))

        A record of up to 20 errors is stored in the power supply’s error queue.

        :rtype: list
        """
        # TODO: Convert to List
        errors = self.instrument.query("SYSTem:ERRor?")
        return errors
